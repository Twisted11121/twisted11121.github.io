TextView & Button

XML 
<Button android:id="@+id/btn" android:text="Click" />
Java 
TextView txt = findViewById(R.id.textView);
Button btn = findViewById(R.id.btn);

btn.setOnClickListener(v -> {
    txt.setText("Button was clicked!");
});

RadioGroup & Buttons

Java 
RadioGroup group = findViewById(R.id.radioGroup);
group.setOnCheckedChangeListener((rg, checkedId) -> {
    RadioButton selected = findViewById(checkedId);
    String text = selected.getText().toString();
});

SeekBar

Java 
SeekBar sb = findViewById(R.id.seekBar);
sb.setOnSeekBarChangeListener(new SeekBar.OnSeekBarChangeListener() {
    @Override
    public void onProgressChanged(SeekBar s, int progress, boolean fromUser) {
        // progress is current value
    }
    @Override public void onStartTrackingTouch(SeekBar s) {}
    @Override public void onStopTrackingTouch(SeekBar s) {}
});

CalendarView to String

Java 
CalendarView cv = findViewById(R.id.calendarView);
cv.setOnDateChangeListener((view, year, month, day) -> {
    // Correcting month (0-11) to (1-12)
    String date = day + "/" + (month + 1) + "/" + year;
    Toast.makeText(this, "Selected: " + date, Toast.LENGTH_SHORT).show();
});

WebView

Manifest 
<uses-permission android:name="android.permission.INTERNET" />
Java 
WebView web = findViewById(R.id.webView);
web.setWebViewClient(new WebViewClient()); // Open links in app
web.getSettings().setJavaScriptEnabled(true);
web.loadUrl("https://www.google.com");

Menu Inflator

Java 
@Override
public boolean onCreateOptionsMenu(Menu menu) {
    getMenuInflater().inflate(R.menu.my_menu, menu);
    return true;
}

@Override
public boolean onOptionsItemSelected(MenuItem item) {
    int id = item.getItemId();
    if (id == R.id.action_settings) return true;
    return super.onOptionsItemSelected(item);
}

SharedPreferences (Local Storage)

Java 
// Saving Data
SharedPreferences sp = getSharedPreferences("UserPrefs", MODE_PRIVATE);
sp.edit().putString("name", "Alex").apply();

// Loading Data
String name = sp.getString("name", "Default Name");

BroadcastReceiver (System Events)

Java 
BroadcastReceiver br = new BroadcastReceiver() {
    @Override
    public void onReceive(Context context, Intent intent) {
        if (Intent.ACTION_BATTERY_LOW.equals(intent.getAction())) {
            // Do something
        }
    }
};
// Dynamic Registration
IntentFilter filter = new IntentFilter(Intent.ACTION_BATTERY_LOW);
registerReceiver(br, filter);
